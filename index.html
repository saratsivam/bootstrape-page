Getting Started

Follows Cartesian coordinate system & Normal units.

In the below example, graphics is initialzed with
{
target: '#canvas',
showGrid:true, //optional
unit:50 // 1 unit is equils to 50 pixes
x:2,y:1, // co-ordinates of the center from bottom left of the target element
}

  • Code
  • Result
/* Create an instance of Graphics */
var g = GUtil.createGraphis({
    target: '#canvas',showGrid:true,x:2,y:1,unit:50	
});


/* Create a rectagle */
var rect = g.create('Rectangle',{x:3,y:2,w:2,h:1});

/* draw rectangle */
g.draw(rect);

Objects Pre defined

Pre defined objects include mathematical objects like Line, Rectangle, Circle, Sphere and
Path: to draw set of 2d points

var rect = g.create('Rectangle',{x:2,y:3,strokeStyle:'green'});
var circle = g.create('Circle',{x:3,y:1.5, r:1, fillStyle:'yellow',showFill:true});
var line = g.create('Line',{x:1,y:2,x2:5,y2:3,strokeStyle:'blue'});
var polarLine = g.create('Line',{type:'polar',x:4,y:4,r:3,angle:45,strokeStyle:'red'});
var sphere = g.create('Sphere',{x:6,y:2,r:0.5,color:'blue'});

/* method chaining */
g.draw(rect).draw(circle);

/* drawArray*/
g.drawArray([line,polarLine,sphere]);

Running Simple Animation

Create and run loop

var loopId= g.loop(function(t){
//code.....
}, time); //time is optional

Stop/Start a loop
g.startLoop(loopId);
g.stopLoop(loopId);
g.clearLoops(); //to stop all the running loops

  • Code
  • Result
/* Create an object instace with vx, vy being the components of velocities */
var rect = g.create('Rectangle',{vx:5,vy:2.3,strokeStyle:'red'});

/* Creating a time loop */
g.loop(function(t){
	/* rotate rectale in time*/
	rect.angle = t/2; 
	
	/* draw the rectanle as well as update its position */
	g.draw(rect.move());

});

Grouping the Objects for Complex Animations

Layer: a container to hold all the objects
we can apply the transformations like rotate, scale, etcs on layer

  • Code
  • Result
/*creating all necessary objects to creat a piston*/
var c = g.create('Circle',{y:-1,r:0.5});	 
var l1 = g.create('Line',{type:'polar',y:-1,r:0.5});	  
var l2 = g.create('Line',{x2:0,r:2});
var r = g.create('Rectangle',{x:0,w:2,h:1,showFill:true});	  
var s1 = g.create('Sphere',{r:0.2});
var s2 = g.create('Sphere',{r:0.2});

/*create a layer to contain all the piston parts*/
var layer = g.create('Layer',{w:2,h:5,scaleX:1,scaleY:1});
layer.add(c).add(s1).add(s2).add(l1).add(l2).add(r);

g.loop(function(t){			
	g.clear();						
	var phase = t*5;				
	buildPiston(phase,2.5);				
	buildPiston(phase+90,5);	
	buildPiston(phase+180,7.5);
	buildPiston(phase+270,10);			
});	

/* drawing the piston for the given phase angle and at position x on the grid*/	
function buildPiston(phase,x){
	l1.angle=phase;	
	s1.position(l1.position2());
	l2.position(l1.position2());			
	l2.y2 = Math.sqrt(l2.r*l2.r-(l2.x2-l2.x)*(l2.x2-l2.x))+l2.y;
	s2.position(l2.position2());			
	r.y = l2.y2;				
	layer.x=x;layer.y=4;/* translate the whole layer to the position x,y*/		
	g.draw(layer);	
}
